-- Loading logic (unchanged)
local progress = 0
local stuckValue = 99.8
local didFakeFail = false
local didRetry = false
local didSuccessText = false

local function updateText(p)
    if p < 10 then
        LoadingText.Text = "Preparing environment..."
    elseif p < 25 then
        LoadingText.Text = "Loading UI components..."
    elseif p < 40 then
        LoadingText.Text = "Loading core logic..."
    elseif p < 55 then
        LoadingText.Text = "Initializing services..."
    elseif p < 70 then
        LoadingText.Text = "Validating game session..."
    elseif p < 80 then
        if not didFakeFail then
            LoadingText.Text = "Attach failed, rolling back changes..."
            didFakeFail = true
        else
            LoadingText.Text = "Re-applying hook configuration..."
        end
    elseif p < 90 then
        if not didRetry then
            LoadingText.Text = "Retrying injection (1/2)..."
            didRetry = true
        else
            LoadingText.Text = "Retrying injection (2/2)..."
        end
    elseif p < stuckValue then
        if not didSuccessText then
            LoadingText.Text = "Successfully attached, finalizing loader..."
            didSuccessText = true
        else
            LoadingText.Text = "Finalizing loader..."
        end
    end
end

local TOTAL_DURATION = math.random(420, 600)
local startTime = os.clock()

task.spawn(function()
    while true do
        local elapsed = os.clock() - startTime
        local t = math.clamp(elapsed / TOTAL_DURATION, 0, 1)

        local eased = 1 - (1 - t) ^ 2
        local targetProgress = stuckValue * eased
        if targetProgress <= progress then
            targetProgress = math.min(progress + 0.02, stuckValue)
        end

        progress = math.clamp(targetProgress, 0, stuckValue)
        updateText(progress)

        PercentText.Text = formatPercent(progress)
        TweenService:Create(
            ProgressBar,
            TweenInfo.new(0.25, Enum.EasingStyle.Sine, Enum.EasingDirection.Out),
            { Size = UDim2.new(progress / 100, 0, 1, 0) }
        ):Play()

        if t >= 1 then break end
        task.wait(0.15 + math.random() * 0.2)
    end

    progress = stuckValue
    updateText(progress)
    PercentText.Text = formatPercent(progress)
    TweenService:Create(
        ProgressBar,
        TweenInfo.new(0.4, Enum.EasingStyle.Sine, Enum.EasingDirection.Out),
        { Size = UDim2.new(progress / 100, 0, 1, 0) }
    ):Play()
end)
